 19 /**
 20  * @class dijon.System
 21  * @constructor
 22 */
 23 dijon.System = function(){
 24 	/** @private */
 25 	this._mappings = {};
 26 
 27 	/** @private */
 28 	this._outlets = {};
 29 
 30     /** @private */
 31     this._handlers = {};
 32 
 33     /** @private */
 34     this._callbacks = {};
 35 
 36     /**
 37      * When <code>true</code> injections are made only if an object has a property with the mapped outlet name.<br/>
 38      * <strong>Set to <code>false</code> at own risk</strong>, may have quite undesired side effects.
 39      * @example
 40      * system.strictInjections = true
 41      * var o = {};
 42      * system.mapSingleton( 'userModel', UserModel );
 43      * system.mapOutlet( 'userModel' );
 44      * system.injectInto( o );
 45      *
 46      * //o is unchanged
 47      *
 48      * system.strictInjections = false;
 49      * system.injectInto( o );
 50      *
 51      * //o now has a member 'userModel' which holds a reference to the singleton instance
 52      * //of UserModel
 53      * @property
 54      * @type Boolean
 55      * @default true
 56      */
 57     this.strictInjections = true;
 58 
 59     /**
 60      * Enables the automatic mapping of outlets for mapped values, singletons and classes
 61      * When this is true any value, singleton or class that is mapped will automatically be mapped as a global outlet
 62      * using the value of <code>key</code> as outlet name
 63      *
 64      * @example
 65      * var o = {
 66      *     userModel : undefined; //inject
 67      * }
 68      * system.mapSingleton( 'userModel', UserModel );
 69      * system.injectInto( o ):
 70      * //o.userModel now holds a reference to the singleton instance of UserModel
 71      * @property
 72      * @type Boolean
 73      * @default false
 74      */
 75     this.autoMapOutlets = false;
 76 };//dijon.System
 77 
 78 dijon.System.prototype = {
 79 
 80 	/**
 81 	 * @private
 82 	 * @param {Class} clazz
 83 	 */
 84 	_createAndSetupInstance : function( key, clazz ){
 85 		var instance = new clazz();
 86 		this.injectInto( instance, key );
 87 		if( "setup" in instance ) instance.setup.call( instance );
 88 		return instance;
 89 	},
 90 
 91 	/**
 92 	 * @private
 93 	 * @param {String} key
 94 	 * @param {Boolean} overrideRules
 95 	 * @return {Object}
 96 	 */
 97 	_retrieveFromCacheOrCreate : function( key, overrideRules ){
 98         if( overrideRules == undefined ) overrideRules = false;
 99         if( this._mappings.hasOwnProperty( key ) ){
100             var config = this._mappings[ key ];
101             var output = null;
102             if( !overrideRules && config.isSingleton ){
103                 if( config.object == null  ){
104                     config.object = this._createAndSetupInstance( key, config.clazz );
105                 }
106                 output = config.object;
107             }else{
108                 if( config.clazz ) output = this._createAndSetupInstance( key, config.clazz );
109             }
110         }else{
111             throw new Error( this.fqn + " is missing a rule for " + key );
112         }
113 		return output
114 	},
115 
116 
117 	/**
118 	 * defines <code>outletName</code> as an injection point in <code>targetKey</code>for the object mapped to <code>sourceKey</code>
119      * @example
120      * var o = {
121      *     user : undefined //inject
122      * }
123      * system.mapValue( 'o', o );
124      * system.mapSingleton( 'userModel', UserModel );
125      * system.mapOutlet( 'userModel', 'o', 'user' );
126      *
127      * var obj = system.getInstance( 'o' );
128      * //obj.user holds a reference to the singleton instance of UserModel
129      *
130      * @example
131      * var o = {
132      *      userModel : undefined //inject
133      * }
134      * system.mapValue( 'o', o );
135      * system.mapSingleton( 'userModel', UserModel );
136      * system.mapOutlet( 'userModel', 'o' );
137      *
138      * var obj = system.getInstance( 'o' );
139      * //obj.userModel holds a reference to the singleton instance of UserModel
140      *
141      * @example
142      *
143      * var o = {
144      *      userModel : undefined
145      * }
146      *
147      * system.mapSingleton( 'userModel', UserModel );
148      * system.mapOutlet( 'userModel' );
149      *
150      * system.injectInto( o );
151      *
152      * //o.userModel holds a reference to the singleton instance of userModel
153      *
154      * @param {String} sourceKey the key mapped to the object that will be injected
155      * @param {String} [targetKey='global'] the key the outlet is assigned to.
156 	 * @param {String} [outletName=source] the name of the property used as an outlet.<br/>
157      * @see dijon.System#unmapOutlet
158 	 */
159 	mapOutlet : function( sourceKey, targetKey, outletName ){
160         if( targetKey == undefined ) targetKey = "global";
161         if( outletName == undefined ) outletName = source;
162         if( ! this._outlets.hasOwnProperty( targetKey ) ) this._outlets[ targetKey ] = {};
163 		this._outlets[ targetKey ][ outletName ] = sourceKey
164 	},
165 
166 	/**
167 	 * Create (if possible) or retrieve an instance of the class mapped to <code>key</code>
168 	 * @param {Object} key
169 	 * @return {Object}
170 	 */
171 	getInstance : function( key ){
172 		return this._retrieveFromCacheOrCreate( key );
173 	},
174 
175 	/**
176 	 * When asked for an instance of the class <code>whenAskedFor</code> or object <code>whenAskedFor</code> inject the instance <code>useValue</code>.
177 	 * @param {String} key
178 	 * @param {Object} useValue
179 	 */
180 	mapValue : function( key, useValue ){
181 		this._mappings[ key ]= {
182             clazz : null,
183             object : useValue,
184             isSingleton : true
185 		}
186         if( this.autoMapOutlets ) this.mapOutlet( key );
187 	},
188 
189 	/**
190 	 * Does a rule exist to satsify such a request?
191 	 * @param {String} key
192 	 * @return {Boolean}
193 	 */
194 	hasMapping : function( key ){
195 		return this._mappings.hasOwnProperty( key );
196 	},
197 
198 	/**
199 	 * When asked for an instance of the class <code>whenAskedFor</code> or for object <code>whenAskedFor</code> inject a <strong>new</strong> instance of <code>instantiateClass</code>.
200 	 * @param {String} key
201 	 * @param {Class} clazz
202 	 */
203 	mapClass : function( key, clazz ){
204 		this._mappings[ key ]= {
205 				clazz : clazz,
206 				object : null,
207 				isSingleton : false
208         }
209         if( this.autoMapOutlets ) this.mapOutlet( key );
210 	},
211 
212     /**
213      * When asked for an instance of the class <code>whenAskedFor</code> or object <code>whenAskedFor</code> inject an instance of <code>useSingletonOf</code>.
214      * @param {String} key
215      * @param {Class} clazz
216      */
217     mapSingleton : function( key, clazz ){
218         this._mappings[ key ] = {
219             clazz : clazz,
220             object : null,
221             isSingleton : true
222         }
223         if( this.autoMapOutlets ) this.mapOutlet( key );
224     },
225 
226 	/**
227 	 * create an instance of the class mapped to <code>keyOrClass</code> and fulfill it's mapped dependencies<br/>
228 	 * <strong>WILL ALWAYS CREATE A NEW INSTANCE</strong>, even if <code>keyOrClass</code> was mapped otherwise or
229 	 * <strong>even when <code>keyOrClass</code> was not mapped</code></strong>.
230 	 * @param {Class} keyOrClass
231 	 * @return {Object}
232 	 */
233 	instantiate : function( key ){
234 		return this._retrieveFromCacheOrCreate( key, true );
235 	},
236 
237 	/**
238 	 * Perform an injection into an object, satisfying all it's dependencies, using the outlets as configured for
239      * <code>key</code>
240 	 * @param {Object} instance
241      * @param {String} [key]
242 	 */
243 	injectInto : function( instance, key ){
244         var o = [];
245         if( this._outlets.hasOwnProperty( 'global' ) ) o.push( this._outlets[ 'global' ] );
246         if( key != undefined && this._outlets.hasOwnProperty( key ) ) o.push( this._outlets[ key ] );
247         for( var i in o ){
248             var l = o [ i ];
249             for( var outlet in l ){
250                 var source = l[ outlet ];
251                 //must be "in" [!]
252                 if( ! this.strictInjections || outlet in instance ) instance[ outlet ] = this.getInstance( source );
253             }
254         }
255 
256         return instance;
257 	},
258 
259 	/**
260 	 * Remove a rule from the System
261 	 * @param {String} key
262 	 */
263 	unmap : function( key ){
264 		delete this._mappings[ key ];
265 	},
266 
267 	/**
268 	 * removes an injection point mapping for a given class mapped to <code>key</code>
269 	 * @param {Object} key
270 	 * @param {String} propertyName MUST BE STRING
271 	 * @see dijon.System#addOutlet
272 	 */
273 	unmapOutlet : function( target, outlet ){
274 		delete this._outlets[ target ][ outlet ];
275 	},
276 
277     /**
278      *
279      * @param {String} eventName
280      * @param {String} key
281      * @param {String|Function} [handler=eventName]
282      * @param {Boolean} [oneShot=false]
283      * @param {Boolean} [passEvent=false]
284      */
285     mapHandler : function( eventName, key, handler, oneShot, passEvent ){
286         if( handler == undefined ) handler = eventName;
287         if( oneShot == undefined ) oneShot = false;
288         if( passEvent == undefined ) passEvent = false;
289         if( ! this._handlers.hasOwnProperty( eventName ) ){
290             this._handlers[ eventName ] = {};
291         }
292         this._handlers[ eventName ][ key ] = {
293             handler : handler,
294             oneShot: oneShot,
295             passEvent: passEvent
296         };
297     },
298 
299     /**
300      *
301      * @param {String} eventName
302      * @param {String} key
303      */
304     unmapHandler : function( eventName, key  ){
305         if( this._handlers.hasOwnProperty( eventName ) && this._handlers[ eventName ].hasOwnProperty( key ) ){
306             delete this._handlers[ eventName ][ key ];
307         }
308     },
309 
310     /**
311      *
312      * @param {String} eventName
313      * @param {Function} callback
314      * @param {Boolean} [oneShot=false]
315      * @param {Boolean} [passEvent=false]
316      */
317     addCallback : function( eventName, callback, oneShot, passEvent ){
318         if( oneShot == undefined ) oneShot = false;
319         if( ! this._callbacks.hasOwnProperty( eventName ) ){
320             this._callbacks[ eventName ] = [];
321         }
322         this._callbacks[ eventName ].push( {
323             callback : callback,
324             oneShot : oneShot,
325             passEvent: passEvent
326         } );
327     },
328 
329     /**
330      *
331      * @param {String} eventName
332      * @param {Function} callback
333      */
334     removeCallback : function( eventName, callback ){
335         if( this._callbacks.hasOwnProperty( eventName ) ){
336             var configs = this._callbacks[ eventName ];
337             for( var i in configs ){
338                 var config = configs[ i ];
339                 if( config.callback === callback ) configs.splice( i, 1 );
340             }
341         }
342     },
343 
344     /**
345      *
346      * @param {String} eventName
347      */
348     notify : function( eventName ){
349         var argsWithEvent = Array.prototype.slice.call( arguments );
350         var argsClean = argsWithEvent.slice(1);
351         if( this._handlers.hasOwnProperty( eventName ) ){
352             var handlers = this._handlers[ eventName ];
353             for( var key in handlers ){
354                 var config = handlers[ key ];
355                 var instance = this.getInstance( key );
356                 var handler;
357                 if( typeof config.handler == "string" ){
358                     handler = instance[ config.handler ];
359                 }else{
360                     handler = config.handler;
361                 }
362                 if( config.oneShot ) this.unmapHandler( eventName, key );
363 
364                 if( config.passEvent ) handler.apply( instance, argsWithEvent );
365                 else handler.apply( instance, argsClean );
366             }
367         }
368         if( this._callbacks.hasOwnProperty( eventName ) ){
369             var callbacks = this._callbacks[ eventName ];
370             for( var i in callbacks ){
371                 var config = callbacks[ i ];
372                 if( config.oneShot ) callbacks.splice( i, 1 );
373 
374                 if( config.passEvent ) config.callback.apply( null, argsWithEvent );
375                 else config.callback.apply( null, argsClean );
376             }
377         }
378     }
379 
380 };//dijon.System.prototype
