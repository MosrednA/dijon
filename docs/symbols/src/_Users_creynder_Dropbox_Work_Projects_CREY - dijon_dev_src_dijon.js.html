 19 /**
 20  * @class dijon.System
 21  * @constructor
 22 */
 23 dijon.System = function(){
 24 	/** @private */
 25 	this._mappings = {};
 26 
 27 	/** @private */
 28 	this._outlets = {};
 29 
 30     /** @private */
 31     this._handlers = {};
 32 
 33     /**
 34      * When <code>true</code> injections are made only if an object has a property with the mapped outlet name.<br/>
 35      * <strong>Set to <code>false</code> at own risk</strong>, may have quite undesired side effects.
 36      * @example
 37      * system.strictInjections = true
 38      * var o = {};
 39      * system.mapSingleton( 'userModel', UserModel );
 40      * system.mapOutlet( 'userModel' );
 41      * system.injectInto( o );
 42      *
 43      * //o is unchanged
 44      *
 45      * system.strictInjections = false;
 46      * system.injectInto( o );
 47      *
 48      * //o now has a member 'userModel' which holds a reference to the singleton instance
 49      * //of UserModel
 50      * @property
 51      * @type Boolean
 52      * @default true
 53      */
 54     this.strictInjections = true;
 55 
 56     /**
 57      * Enables the automatic mapping of outlets for mapped values, singletons and classes
 58      * When this is true any value, singleton or class that is mapped will automatically be mapped as a global outlet
 59      * using the value of <code>key</code> as outlet name
 60      *
 61      * @example
 62      * var o = {
 63      *     userModel : undefined; //inject
 64      * }
 65      * system.mapSingleton( 'userModel', UserModel );
 66      * system.injectInto( o ):
 67      * //o.userModel now holds a reference to the singleton instance of UserModel
 68      * @property
 69      * @type Boolean
 70      * @default false
 71      */
 72     this.autoMapOutlets = false;
 73 };//dijon.System

 77 	/**
 78 	 * @private
 79 	 * @param {Class} clazz
 80 	 */
 81 	_createAndSetupInstance : function( key, clazz ){
 82 		var instance = new clazz();
 83 		this.injectInto( instance, key );
 84 		if( "setup" in instance ) instance.setup.call( instance );
 85 		return instance;
 86 	},

 88 	/**
 89 	 * @private
 90 	 * @param {String} key
 91 	 * @param {Boolean} overrideRules
 92 	 * @return {Object}
 93 	 */
 94 	_retrieveFromCacheOrCreate : function( key, overrideRules ){
 95         if( overrideRules == undefined ) overrideRules = false;
 96         if( this._mappings.hasOwnProperty( key ) ){
 97             var config = this._mappings[ key ];
 98             var output = null;
 99             if( !overrideRules && config.isSingleton ){
100                 if( config.object == null  ){
101                     config.object = this._createAndSetupInstance( key, config.clazz );
102                 }
103                 output = config.object;
104             }else{
105                 if( config.clazz ) output = this._createAndSetupInstance( key, config.clazz );
106                 else{
107 
108                 }
109             }
110         }else{
111             throw new Error( 1020 );
112         }
113 		return output
114 	},

117 	/**
118 	 * defines <code>outletName</code> as an injection point in <code>targetKey</code>for the object mapped to <code>sourceKey</code>
119      * @example
120      * var o = {
121      *     user : undefined //inject
122      * }
123      * system.mapValue( 'o', o );
124      * system.mapSingleton( 'userModel', UserModel );
125      * system.mapOutlet( 'userModel', 'o', 'user' );
126      *
127      * var obj = system.getInstance( 'o' );
128      * //obj.user holds a reference to the singleton instance of UserModel
129      *
130      * @example
131      * var o = {
132      *      userModel : undefined //inject
133      * }
134      * system.mapValue( 'o', o );
135      * system.mapSingleton( 'userModel', UserModel );
136      * system.mapOutlet( 'userModel', 'o' );
137      *
138      * var obj = system.getInstance( 'o' );
139      * //obj.userModel holds a reference to the singleton instance of UserModel
140      *
141      * @example
142      *
143      * var o = {
144      *      userModel : undefined
145      * }
146      *
147      * system.mapSingleton( 'userModel', UserModel );
148      * system.mapOutlet( 'userModel' );
149      *
150      * system.injectInto( o );
151      *
152      * //o.userModel holds a reference to the singleton instance of userModel
153      *
154      * @param {String} sourceKey the key mapped to the object that will be injected
155      * @param {String} [targetKey='global'] the key the outlet is assigned to.
156 	 * @param {String} [outletName=sourceKey] the name of the property used as an outlet.<br/>
157      * @return {dijon.System}
158      * @see dijon.System#unmapOutlet
159 	 */
160 	mapOutlet : function( sourceKey, targetKey, outletName ){
161 	    if( sourceKey == undefined ) throw new Error( 1010 );
162         if( targetKey == undefined ) targetKey = "global";
163         if( outletName == undefined ) outletName = sourceKey;
164         if( ! this._outlets.hasOwnProperty( targetKey ) ) this._outlets[ targetKey ] = {};
165 		this._outlets[ targetKey ][ outletName ] = sourceKey;
166 
167         return this;
168 	},
169 
170 	/**
171 	 * Retrieve (or create) the object mapped to <code>key</code>
172      * @example
173      * system.mapValue( 'foo', 'bar' );
174      * var b = system.getObject( 'foo' ); //now contains 'bar'
175 	 * @param {Object} key
176 	 * @return {Object}
177 	 */
178 	getObject : function( key ){
179 	    if( key == undefined ) throw new Error( 1010 )
180 		return this._retrieveFromCacheOrCreate( key );
181 	},
182 
183 	/**
184 	 * Maps <code>useValue</code> to <code>key</code>
185      * @example
186      * system.mapValue( 'foo', 'bar' );
187      * var b = system.getObject( 'foo' ); //now contains 'bar'
188 	 * @param {String} key
189 	 * @param {Object} useValue
190      * @return {dijon.System}
191 	 */
192 	mapValue : function( key, useValue ){
193 	    if( key == undefined ) throw new Error( 1010 );
194 		this._mappings[ key ]= {
195             clazz : null,
196             object : useValue,
197             isSingleton : true
198 		}
199         if( this.autoMapOutlets ) this.mapOutlet( key );
200         return this;
201 	},
202 
203 	/**
204 	 * Returns whether the key is mapped to an object
205      * @example
206      * system.mapValue( 'foo', 'bar' );
207      * var isMapped = system.hasMapping( 'foo' );
208 	 * @param {String} key
209 	 * @return {Boolean}
210 	 */
211 	hasMapping : function( key ){
212 	    if( key == undefined ) throw new Error( 1010 );
213 		return this._mappings.hasOwnProperty( key );
214 	},
215 
216 	/**
217      * Maps <code>clazz</code> as a factory to <code>key</code>
218      * @example
219      * var SomeClass = function(){
220      * }
221      * system.mapClass( 'o', SomeClass );
222      *
223      * var s1 = system.getInstance( 'o' );
224      * var s2 = system.getInstance( 'o' );
225      *
226      * //s1 and s2 reference two different instances of SomeClass
227      *
228 	 * @param {String} key
229 	 * @param {Function} clazz
230      * @return {dijon.System}
231 	 */
232 	mapClass : function( key, clazz ){
233 	    if( key == undefined ) throw new Error( 1010 );
234 	    if( key == undefined ) throw new Error( 1010 );
235 		this._mappings[ key ]= {
236 				clazz : clazz,
237 				object : null,
238 				isSingleton : false
239         }
240         if( this.autoMapOutlets ) this.mapOutlet( key );
241         return this;
242 	},
243 
244     /**
245      * Maps <code>clazz</code> as a singleton factory to <code>key</code>
246      * @example
247      * var SomeClass = function(){
248      * }
249      * system.mapSingleton( 'o', SomeClass );
250      *
251      * var s1 = system.getInstance( 'o' );
252      * var s2 = system.getInstance( 'o' );
253      *
254      * //s1 and s2 reference the same instance of SomeClass
255      *
256      * @param {String} key
257      * @param {Function} clazz
258      * @return {dijon.System}
259      */
260     mapSingleton : function( key, clazz ){
261         if( key == undefined ) throw new Error( 1010 );
262         if( clazz == undefined ) throw new Error( 1010 );
263         this._mappings[ key ] = {
264             clazz : clazz,
265             object : null,
266             isSingleton : true
267         }
268         if( this.autoMapOutlets ) this.mapOutlet( key );
269         return this;
270     },
271 
272 	/**
273      * Force instantiation of the object mapped to <code>key</code>, whether it was mapped as a singleton or not.
274 	 * @param {String} key
275 	 * @return {Object}
276 	 */
277 	instantiate : function( key ){
278         if( key == undefined ) throw new Error( 1010 );
279 		return this._retrieveFromCacheOrCreate( key, true );
280 	},
281 
282 	/**
283 	 * Perform an injection into an object's mapped outlets, satisfying all it's dependencies
284 	 * @param {Object} instance
285      * @param {String} [key]
286      * @return {dijon.System}
287 	 */
288 	injectInto : function( instance, key ){
289         if( instance == undefined ) throw new Error( 1010 );
290         var o = [];
291         if( this._outlets.hasOwnProperty( 'global' ) ) o.push( this._outlets[ 'global' ] );
292         if( key != undefined && this._outlets.hasOwnProperty( key ) ) o.push( this._outlets[ key ] );
293         for( var i in o ){
294             var l = o [ i ];
295             for( var outlet in l ){
296                 var source = l[ outlet ];
297                 //must be "in" [!]
298                 if( ! this.strictInjections || outlet in instance ) instance[ outlet ] = this.getObject( source );
299             }
300         }
301 
302         return this;
303 	},
304 
305 	/**
306 	 * Remove the mapping of <code>key</code> from the system
307 	 * @param {String} key
308      * @return {dijon.System}
309 	 */
310 	unmap : function( key ){
311         if( key == undefined ) throw new Error( 1010 );
312 		delete this._mappings[ key ];
313 
314         return this;
315 	},
316 
317 	/**
318 	 * removes an injection point mapping for a given object mapped to <code>key</code>
319 	 * @param {String} target
320 	 * @param {String} outlet
321      * @return {dijon.System}
322 	 * @see dijon.System#addOutlet
323 	 */
324 	unmapOutlet : function( target, outlet ){
325         if( target == undefined ) throw new Error( 1010 );
326         if( outlet == undefined ) throw new Error( 1010 );
327 		delete this._outlets[ target ][ outlet ];
328 
329         return this;
330 	},
331 
332     /**
333      * maps a handler for an event/route.<br/>
334      * @param {String} eventName/route
335      * @param {String} [key=undefined] If <code>key</code> is <code>undefined</code> the handler will be mapped for all
336      * mapped objects.
337      * @param {String|Function} [handler=eventName] If <code>handler</code> is <code>undefined</code> the value of
338      * <code>eventName</code> will be used as the name of the member holding the reference to the to-be-called function.
339      * <code>handler</code> accepts either a string, which will be used as the name of the member holding the reference
340      * to the to-be-called function, or a direct function reference.
341      * @param {Boolean} [oneShot=false] Defines whether the handler should called exactly ones and then automatically
342      * unmapped
343      * @param {Boolean} [passEvent=false] Defines whether the event object should be passed to the handler or not.
344      * @return {dijon.System}
345      * @see dijon.System#unmapHandler
346      */
347     mapHandler : function( eventName, key, handler, oneShot, passEvent ){
348         if( eventName == undefined ) throw new Error( 1010 );
349         if( key == undefined ) key = 'global';
350         if( handler == undefined ) handler = eventName;
351         if( oneShot == undefined ) oneShot = false;
352         if( passEvent == undefined ) passEvent = false;
353         if( ! this._handlers.hasOwnProperty( eventName ) ){
354             this._handlers[ eventName ] = {};
355         }
356         if( ! this._handlers[eventName].hasOwnProperty( key ) ){
357             this._handlers[eventName][key] = [];
358         }
359         this._handlers[ eventName ][ key ].push( {
360             handler : handler,
361             oneShot: oneShot,
362             passEvent: passEvent
363         } );
364 
365         return this;
366     },
367 
368     /**
369      * Unmaps the handler for a specific event/route.
370      * @param {String} eventName Name of the event/route
371      * @param {String} [key=undefined] If <code>key</code> is <code>undefined</code> the handler is removed from the
372      * global mapping space. (If the same event is mapped globally and specifically for an object, then
373      * only the globally mapped one will be removed)
374      * @param {String | Function} [handler=eventName]
375      * @return {dijon.System}
376      * @see dijon.System#mapHandler
377      */
378     unmapHandler : function( eventName, key, handler  ){
379         if( eventName == undefined ) throw new Error( 1010 );
380         if( key == undefined ) key = 'global';
381         if( handler == undefined ) handler = eventName;
382         if( this._handlers.hasOwnProperty( eventName ) && this._handlers[ eventName ].hasOwnProperty( key ) ){
383             var handlers = this._handlers[ eventName ][ key ];
384             for( var i in handlers ){
385                 var config = handlers[ i ];
386                 if( config.handler == handler ){
387                     handlers.splice( i, 1 );
388                     return;
389                 }
390             }
391         }
392         return this
393     },
394 
395     /**
396      *
397      * @param {String} eventName
398      */
399     notify : function( eventName ){
400         var argsWithEvent = Array.prototype.slice.call( arguments );
401         var argsClean = argsWithEvent.slice(1);
402         if( this._handlers.hasOwnProperty( eventName ) ){
403             var handlers = this._handlers[ eventName ];
404             for( var key in handlers ){
405                 var configs = handlers[ key ];
406                 var instance;
407                 if( key != 'global' ) instance = this.getObject( key );
408                 var toBeDeleted = [];
409                 for( var i = 0, n = configs.length; i < n; i++ ){
410                     var handler;
411                     var config = configs[ i ];
412                     if( instance && typeof config.handler == "string" ){
413                         handler = instance[ config.handler ];
414                     }else{
415                         handler = config.handler;
416                     }
417 
418                     //see deletion below
419                     if( config.oneShot ) toBeDeleted.unshift( i );
420 
421                     if( config.passEvent ) handler.apply( instance, argsWithEvent );
422                     else handler.apply( instance, argsClean );
423                 }
424 
425                 //items should be deleted in reverse order
426                 //either use push above and decrement here
427                 //or use unshift above and increment here
428                 for( var i = 0, n = toBeDeleted.length; i < n; i++ ){
429                     configs.splice( toBeDeleted[ i ], 1 );
430                 }
431             }
432         }
433     }
