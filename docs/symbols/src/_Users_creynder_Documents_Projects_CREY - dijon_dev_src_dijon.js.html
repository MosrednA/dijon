 19 dijon.EventDispatcher = function(){

 27 	this._length = 0;

 31 dijon.EventDispatcher.prototype = {
 32 
 33 	/**
 34 	 * @private
 35 	 * @param {String} eventType
 36 	 * @param {Number} index
 37 	 */
 38 	_removeListenerByIndex : function( eventType, index ){
 39 		this._listeners[ eventType ].splice( index, 1 );
 40 		if( this._listeners[ eventType ].length <= 0 )
 41 			delete this._listeners[ eventType ];
 42 		this._length--;
 43 	},
 44 
 45 	/**
 46 	 * adds a handler to be invoked when an event is dispatched
 47 	 * @param {String} eventType The name of the event to be listened to
 48 	 * @param {Function} listener The handler to be called when the event has been dispatched
 49 	 * @param {Boolean} [oneShot=false] Whether the listener must be called only once
 50 	 * @param {Boolean} [passEvent=false] Whether the event object needs to be passed to the listener
 51 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
 52 	*/
 53 	addListener : function( eventType, listener, oneShot, passEvent ) {
 54 		this.addScopedListener( eventType, listener, undefined, oneShot, passEvent );
 55 		return this;
 56 	},
 57 
 58 	/**
 59 	 * adds a handler to be invoked in a specific <code>scope</code> when an event of <code>eventType</code> is dispatched
 60 	 * @param {String} eventType The name of the event to be listened to
 61 	 * @param {Function} listener The handler to be called when the event has been dispatched
 62 	 * @param {Object} scope The scope in which the listener will be called
 63 	 * @param {Boolean} [oneShot=false] Whether the listener must be called only once
 64 	 * @param {Boolean} [passEvent=false] Whether the event object needs to be passed to the listener
 65 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
 66 	 */
 67 	addScopedListener : function( eventType, listener, scope, oneShot, passEvent ){
 68 		if( oneShot == undefined ) oneShot = false;
 69 		if( passEvent == undefined ) passEvent = false;
 70 		if( this._listeners[ eventType ] == undefined ){
 71 			this._listeners[ eventType ] = [];
 72 		}
 73 		this._listeners[ eventType ].push( {
 74 			scope : scope,
 75 			listener : listener,
 76 			oneShot : oneShot,
 77 			passEvent : passEvent
 78 		} );
 79 		++this._length;
 80 		return this;
 81 	},
 82 
 83 	/**
 84 	 * Checks whether the dispatcher object has registered <code>listener</code> for <code>eventType</code>
 85 	 * @param {String} eventType The name of the event to be listened to
 86 	 * @param {Function} listener The handler to be called when the event has been dispatched
 87 	 * @return {Boolean}
 88 	 */
 89 	hasListener : function( eventType, listener ){
 90 		return this.hasScopedListener( eventType, listener, undefined );
 91 	},
 92 
 93 	/**
 94 	 * Checks whether the dispatcher object has registered <code>listener</code> in <code>scope</code> for <code>eventType</code>
 95 	 * @param {String} eventType The name of the event to be listened to
 96 	 * @param {Function} listener The handler to be called when the event has been dispatched
 97 	 * @param {Object} scope The scope in which the listener will be called
 98 	 * @return {Boolean}
 99 	 */
100 	hasScopedListener : function( eventType, listener, scope ){
101 		if( this._listeners[ eventType ] ){
102 			var listeners = this._listeners[ eventType ];
103 			for( var i = 0, n = listeners.length ; i < n ; i++  ){
104 				var obj = listeners[ i ];
105 				if( obj.listener === listener && obj.scope === scope ){
106 					return true;
107 				}
108 			}
109 		}
110 
111 		return false;
112 	},
113 
114 	/**
115 	 * Removes a <code>listener</code> registered for <code>eventType</code>
116 	 * @param {String} eventType The name of the event to be listened to
117 	 * @param {Function} listener The handler to be called when the event has been dispatched
118 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
119 	 */
120 	removeListener : function( eventType, listener ){
121 		this.removeScopedListener( eventType, listener, undefined );
122 		return this;
123 	},
124 
125 	/**
126 	 * Removes a <code>listener</code> in <code>scope</code> registered for <code>eventType</code>
127 	 * @param {String} eventType The name of the event to be listened to
128 	 * @param {Function} listener The handler to be called when the event has been dispatched
129 	 * @param {Object} scope The scope in which the listener will be called
130 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
131 	 */
132 	removeScopedListener : function( eventType, listener, scope ) {
133 		for ( var i = 0 ; this._listeners[ eventType ] && i < this._listeners[ eventType ].length ;  ){
134 			var obj = this._listeners[ eventType ][ i ];
135 			if( obj.listener == listener && obj.scope === scope ){
136 				this._removeListenerByIndex( eventType, i );
137 			}else{
138 				i++;
139 			}
140 		}
141 		return this;
142 	},
143 
144 	/**
145 	 * dispatches an event with any number of arguments
146 	 * @param {Object|String} event The event object or event type
147 	 * @param {String} event.type if <code>event</code> is of type <code>Object</code> it requires a <code>type</code> property set to the name of the event that will be listened to.
148 	 * @param {...} [...=undefined] Any number of parameters
149 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
150 	 */
151 	dispatchEvent : function( event ){
152 		if (typeof event == "string"){
153 			event = { type: event };
154 		}
155 		if (!event.target){
156 			event.target = this;
157 		}
158 		var args = [];
159 		for( var i = 1, n = arguments.length; i < n ; i++ ){
160 			args.push( arguments[ i ] );
161 		}
162 
163 		if( this._listeners[ event.type ] ){
164 			for( var i = 0 ; this._listeners[ event.type ] && i < this._listeners[ event.type ].length ;  ){
165 				var obj = this._listeners[ event.type ][ i ];
166 				if( obj.oneShot ){
167 					this._removeListenerByIndex( event.type, i );
168 				}else{
169 					i++;
170 				}
171 				if( obj.passEvent ){
172 					args.unshift( event );
173 				}
174 				obj.listener.apply( obj.scope, args );
175 			}
176 		}
177 
178 		return this;
179 	},
180 
181 	/**
182 	 * removes all event listeners
183 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
184 	 */
185 	removeAllListeners : function(){
186 	   this._listeners = {};
187 	   this._length = 0;
188 
189 	   return this;
190 	},
191 
192 	/**
193 	 * Returns the number of listeners
194 	 * @return {Number} The number of listeners
195 	 */
196 	length : function(){
197 		return this._length;
198 	}
199 
200 };//dijon.EventDispatcher.prototype

223 dijon.Dictionary.prototype = {
224 	/**
225 	 * @private
226 	 * @param key
227 	 * @return index
228 	 */
229 	_getIndexByKey : function( key ){
230 		for( var i = 0, n = this._map.length ; i < n ; i++ ){
231 			if( this._map[ i ].key === key ) return i;
232 		}
233 
234 		return -1;
235 	},
236 
237 	/**
238 	 * Maps <code>value</code> to <code>key</code>
239 	 * @param {Object} key
240 	 * @param {Object} value
241 	 * @return {dijon.Dictionary} the Dictionary instance
242 	 */
243 	add : function( key, value ){
244 		var index = this._getIndexByKey( key );
245 		if( index < 0 ){
246 			this._map.push( {
247 				key : key,
248 				value : value
249 			});
250 		}else{
251 			this._map[ index ] = {
252 				key : key,
253 				value : value
254 			}
255 		}
256 		return this;
257 	},
258 
259 	/**
260 	 * removes the mapping of the value of <code>key</code>
261 	 * @param {Object} key
262 	 * @return {dijon.Dictionary} the Dictionary instance
263 	 */
264 	remove : function( key ){
265 		var index = this._getIndexByKey( key );
266 		if( index >= 0 ) return this._map.splice( index, 1 ).value;
267 
268 		return this;
269 	},
270 
271 	/**
272 	 * retrieves the value mapped to <code>key</code>
273 	 * @param {Object} key
274 	 * @return {Object} the value mapped to <code>key</code>
275 	 */
276 	getValue : function( key ){
277 		var index = this._getIndexByKey( key );
278 
279 		if( index >= 0 ) return this._map[ index ].value;
280 
281 		return null;
282 	},
283 
284 	/**
285 	 * checks whether a value has been mapped to <code>key</code>
286 	 * @param {Object} key
287 	 * @return {Boolean}
288 	 */
289 	hasValue : function( key ){
290 		var index = this._getIndexByKey( key );
291 		return ( index >= 0 );
292 	}
293 
294 };//dijon.Dictionary.prototype

300 /**
301  @class dijon.Injector
302  @author Camille Reynders - www.creynders.be
303  @constructor
304 */
305 dijon.Injector = function(){
306 	this.qcn = 'dijon.Injector';
307 
308 	/** @private */
309 	this._mappingsByEventType = new dijon.Dictionary();
310 
311 	/** @private */
312 	this._injectionPoints = [];
313 };//dijon.Injector
314 
315 dijon.Injector.prototype = {
316 
317 	/**
318 	 * @private
319 	 * @param {Class} clazz
320 	 */
321 	_createAndSetupInstance : function( clazz ){
322 		var instance = new clazz();
323 		this.injectInto( instance );
324 		if( "setup" in instance ) instance.setup.call( instance );
325 		return instance;
326 	},
327 
328 	/**
329 	 * @private
330 	 * @param {Class} clazz
331 	 * @param {Boolean} overrideSingleton
332 	 * @return {Object}
333 	 */
334 	_retrieveFromCacheOrCreate : function( clazz, overrideSingleton ){
335 		var value = this._mappingsByEventType.getValue( clazz );
336 		var output = null;
337 		if( value ){
338 			//found
339 			if( value.isSingleton && ! overrideSingleton ){
340 				if( value.object == null ){
341 					 value.object = this._createAndSetupInstance( value.clazz );
342 				}
343 				output = value.object;
344 			}else{
345 				output = this._createAndSetupInstance( value.clazz );
346 			}
347 		}else{
348 			throw new Error( this.qcn + " is missing a rule for " + clazz );
349 		}
350 		return output;
351 	},
352 
353 	/**
354 	 * defines <code>propertyName</code> as an injection point for <code>targetClazz</code> to be injected with an instance
355 	 * of <code>sourceClazz</code>.
356 	 * @param {Class} targetClazz the class the injection point is applied to.
357 	 * @param {String} propertyName the <strong>name</strong> of the property used as an injection point.<br/>
358 	 * [!] MUST BE STRING
359 	 * @param {Class} sourceClazz the type of the instance that will be injected
360 	 */
361 	addInjectionPoint : function( targetClazz, propertyName, sourceClazz ){
362 		this._injectionPoints.push( {
363 			target : targetClazz,
364 			property : propertyName,
365 			clazz : sourceClazz
366 		} );
367 	},
368 
369 	/**
370 	 * Create (if possible) or retrieve an instance of the class mapped to <code>clazz</code>
371 	 * @param {Class} clazz
372 	 * @return {Object}
373 	 */
374 	getInstance : function( clazz ){
375 		return this._retrieveFromCacheOrCreate( clazz, false );
376 	},
377 
378 	/**
379 	 * When asked for an instance of the class <code>whenAskedFor<code> or object <code>whenAskedFor</code> inject an instance of <code>whenAskedFor<code>.
380 	 * @param {Class|Object} whenAskedFor
381 	 */
382 	mapSingleton : function( whenAskedFor ){
383 		if( this._mappingsByEventType.hasValue( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
384 		this.mapSingletonOf( whenAskedFor, whenAskedFor );
385 	},
386 
387 	/**
388 	 * When asked for an instance of the class <code>whenAskedFor</code> or object <code>whenAskedFor</code> inject the instance <code>useValue</code>.
389 	 * @param {Class|Object} whenAskedFor
390 	 * @param {Object} useValue
391 	 */
392 	mapValue : function( whenAskedFor, useValue ){
393 		if( this._mappingsByEventType.hasValue( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
394 		this._mappingsByEventType.add(
395 			whenAskedFor,
396 			{
397 				clazz : whenAskedFor,
398 				object : useValue,
399 				isSingleton : true
400 			}
401 		);
402 	},
403 
404 	/**
405 	 * Does a rule exist to satsify such a request?
406 	 * @param {Class} whenAskedFor
407 	 * @return {Boolean}
408 	 */
409 	hasMapping : function( whenAskedFor ){
410 		return this._mappingsByEventType.hasValue( whenAskedFor );
411 	},
412 
413 	/**
414 	 * When asked for an instance of the class <code>whenAskedFor</code> or for object <code>whenAskedFor</code> inject a <strong>new</strong> instance of <code>instantiateClass</code>.
415 	 * @param {Class|Object} whenAskedFor
416 	 * @param {Class} instantiateClass
417 	 */
418 	mapClass : function( whenAskedFor, instantiateClass ){
419 		if( this.hasMapping( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
420 		this._mappingsByEventType.add(
421 			whenAskedFor,
422 			{
423 				clazz : instantiateClass,
424 				object : null,
425 				isSingleton : false
426 			}
427 		);
428 	},
429 
430 	/**
431 	 * When asked for an instance of the class <code>whenAskedFor</code> or object <code>whenAskedFor</code> inject an instance of <code>useSingletonOf</code>.
432 	 * @param {Class|Object} whenAskedFor
433 	 * @param {Class} useSingletonOf
434 	 */
435 	mapSingletonOf : function( whenAskedFor, useSingletonOf ){
436 		if( this._mappingsByEventType.hasValue( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
437 		this._mappingsByEventType.add(
438 			whenAskedFor,
439 			{
440 				clazz : useSingletonOf,
441 				object : null,
442 				isSingleton : true
443 			}
444 		);
445 	},
446 
447 	/**
448 	 * create an instance of the class mapped to <code>clazz</code> and fulfill it's mapped dependencies<br/>
449 	 * <strong>WILL ALWAYS CREATE A NEW INSTANCE</strong>, even if <code>clazz</code> was mapped otherwise.
450 	 * @param {Class} clazz
451 	 * @return {Object}
452 	 */
453 	instantiate : function( clazz ){
454 		return this._retrieveFromCacheOrCreate( clazz, true );
455 	},
456 
457 	/**
458 	 * Perform an injection into an object, satisfying all it's dependencies
459 	 * @param {Object} instance
460 	 */
461 	injectInto : function( instance ){
462 		for( var i = 0, n = this._injectionPoints.length ; i < n ; i++ ){
463 			var mapping = this._injectionPoints[ i ];
464 			if( instance && instance instanceof mapping.target && mapping.property in instance )
465 				instance[ mapping.property ] = this.getInstance( mapping.clazz );
466 		}
467 	},
468 
469 	/**
470 	 * Remove a rule from the injector
471 	 * @param {Class|Object} whenAskedFor
472 	 */
473 	unmap : function( whenAskedFor ){
474 		this._mappingsByEventType.remove( whenAskedFor );
475 	},
476 
477 	/**
478 	 * removes an injection point mapping for a given class
479 	 * @param {Function} targetClazz
480 	 * @param {String} propertyName MUST BE STRING
481 	 * @see dijon.Injector#addInjectionPoint
482 	 */
483 	removeInjectionPoint : function( targetClazz, propertyName ){
484 		for( var i = 0, n = this._injectionPoints.length ; i < n ; i++ ){
485 			var point = this._injectionPoints[ i ];
486 			if( point.target == targetClazz && point.property == propertyName ) {
487 				this._injectionPoints.splice( i, 1 );
488 				return;
489 			}
490 		}
491 	}
492 
493 };//dijon.Injector.prototype

499 /**
500  * registers class members as listeners for specific events, before (or after) instantiation.<br/>
501  * Allows for mapping listeners to lazily instantiated objects.<br/>
502  * [!] This class differs substantially from the RobotLegs EventMap both in use and functionality
503  * @class dijon.EventMap
504  * @author Camille Reynders - www.creynders.be
505  * @constructor
506  */
507 dijon.EventMap = function(){
508 	this.qcn = 'dijon.EventMap';
509 
510 	/**
511 	 * @private
512 	 * @type Object
513 	 */
514 	this._mappingsByEventType = {};
515 
516 	/**
517 	 * @private
518 	 * @type Object
519 	 */
520 	this._mappingsNumByClazz = {};
521 
522 	/**
523 	 * @private
524 	 * @type dijon.EventDispatcher
525 	 */
526 	this.dispatcher = undefined; //inject
527 
528 	/**
529 	 * @private
530 	 * @type dijon.Injector
531 	 */
532 	this.injector = undefined; //inject
533 
534 };//dijon.EventMap
535 
536 dijon.EventMap.prototype = {
537 	/**
538 	 * @private
539 	 * @param {Object} event
540 	 */
541 	_handleRuledMappedEvent : function( event ){
542 		var mappingsListForEvent = this._mappingsByEventType[ event.type ];
543 		var args = [];
544 
545 		for( var i = 1, n = arguments.length; i < n ; i++ ){
546 			args.push( arguments[ i ] );
547 		}
548 		for( var i = 0, n = mappingsListForEvent.length; i < n; i++ ){
549 			var obj = mappingsListForEvent[i];
550 			if( this.injector.hasMapping( obj.clazz ) ){
551 				if( obj.oneShot )
552 					this.removeRuledMapping( event.type, obj.clazz, obj.handler );
553 				var instance = this.injector.getInstance( obj.clazz );
554 				if( obj.passEvent )
555 					args.unshift( event );
556 				obj.handler.apply( instance, args );
557 			}else{
558 				//injector mapping has been deleted, but
559 				//eventMap mapping not
560 				//TODO: remove or throw error?
561 			}
562 		}
563 	},
564 
565 	/**
566 	 * @private
567 	 * @param {String} eventType
568 	 * @param {Class} clazz
569 	 * @param {Function} handler
570 	 */
571 	_removeRuledMappingAndUnmapFromInjectorIfNecessary : function( eventType, clazz, handler ){
572 		this.removeRuledMapping( eventType, clazz, handler );
573 		var mappingsNum = this._mappingsNumByClazz[ clazz ] || 0;
574 		if( mappingsNum <= 0 )
575 			this.injector.unmap( clazz );
576 	},
577 
578 	/**
579 	 * maps <code>handler</code> as a listener for <code>eventType</code> to be called as a member of a <code>clazz</code>
580 	 * instance. The instance will be created according to the rule defined for <code>clazz</code> in injector.
581 	 * <br/>[!] requires <code>clazz</code> is already ruled by the injector
582 	 * @see dijon.Injector
583 	 * @param {String} eventType The name of the event to be listened to
584 	 * @param {Class} clazz
585 	 * @param {Function} handler
586 	 * @param {Function} oneShot
587 	 * @param {Boolean} [passEvent=false]
588 	 */
589 	addRuledMapping : function( eventType, clazz, handler, oneShot, passEvent ){
590 		if( ! this._mappingsByEventType[ eventType ] ){
591 			this._mappingsByEventType[ eventType ] = [];
592 			this.dispatcher.addScopedListener( eventType, this._handleRuledMappedEvent, this, false, true );
593 		}
594 
595 		var mappingsNum = this._mappingsNumByClazz[ clazz ] || 0;
596 		this._mappingsNumByClazz[ clazz ] = ++mappingsNum;
597 
598 		this._mappingsByEventType[ eventType ].push( { clazz : clazz, handler : handler, oneShot : oneShot, passEvent: passEvent } );
599 	},
600 
601 	/**
602 	 *
603 	 * @param eventType
604 	 * @param clazz
605 	 * @param handler
606 	 */
607 	removeRuledMapping : function( eventType, clazz, handler ){
608 		var mappingsListForEvent = this._mappingsByEventType[ eventType ];
609 		if( mappingsListForEvent ){
610 			for( var i = 0, n = mappingsListForEvent.length; i < n ; i++ ){
611 				var mapping = mappingsListForEvent[ i ];
612 				if( mapping.clazz === clazz && mapping.handler === handler ){
613 					delete mapping.clazz;
614 					delete mapping.handler;
615 					mapping = null;
616 					mappingsListForEvent.splice( i, 1 );
617 					if( mappingsListForEvent.length <= 0 )
618 						delete mappingsListForEvent[ eventType ];
619 					var mappingsNum = this._mappingsNumByClazz[ clazz ] || 0;
620 					if( mappingsNum <= 0 ){
621 						delete this._mappingsNumByClazz[ clazz ];
622 					}else{
623 						this._mappingsNumByClazz[ clazz ] = --mappingsNum;
624 					}
625 					return true;
626 				}
627 			}
628 		}
629 
630 		return false;
631 	},
632 
633 	/**
634 	 *
635 	 * @param eventType
636 	 * @param clazz
637 	 * @param handler
638 	 * @param {Boolean} [oneShot] default false
639 	 * @param {Boolean} [passEvent] default false
640 	 */
641 	addClassMapping : function( eventType, clazz, handler, oneShot, passEvent ){
642 		if( ! this.injector.hasMapping( clazz ) ){
643 			this.injector.mapClass( clazz, clazz );
644 		}
645 
646 		this.addRuledMapping( eventType, clazz, handler, oneShot, passEvent );
647 	},
648 
649 	removeClassMapping : function( eventType, clazz, handler ){
650 		this._removeRuledMappingAndUnmapFromInjectorIfNecessary( eventType, clazz, handler );
651 	}
652 
653 };//dijon.EventMap.prototype

757  * @constructor
758  */
759 dijon.Context = function(){
760 	this.qcn = 'dijon.Context';
761 
762 	/**
763 	 * @type dijon.Injector
764 	 */
765 	this.injector = undefined;
766 
767 	/**
768 	 * @type dijon.EventMap
769 	 */
770 	this.eventMap = undefined;
771 
772 	/**
773 	 * @type dijon.EventDispatcher
774 	 */
775 	this.eventDispatcher = undefined;
776 
777 	/**
778 	 * @type dijon.CommandMap
779 	 */
780 	this.commandMap = undefined;
781 
782 	this._createInjector();
783 	this._wireAndCreateEventDispatcher();
784 	this._wireAndCreateEventMap();
785 	this._wireActor();
786 	this._wireCommand();
787 	this._wireAndCreateCommandMap();
788 
789 };//dijon.Context
790 
791 dijon.Context.prototype = {
792 
793 	/**
794 	 * @private
795 	 */
796 	_createInjector : function(){
797 		this.injector = new dijon.Injector();
798 		this.injector.mapValue( dijon.Injector, this.injector );
799 	},
800 	/**
801 	 * @private
802 	 */
803 	_wireAndCreateEventDispatcher : function(){
804 		this.injector.mapSingleton( dijon.EventDispatcher );
805 		this.eventDispatcher = this.injector.getInstance( dijon.EventDispatcher );
806 	},
807 	/**
808 	 * @private
809 	 */
810 	_wireAndCreateEventMap : function(){
811 		this.injector.addInjectionPoint( dijon.EventMap, 'dispatcher', dijon.EventDispatcher );
812 		this.injector.addInjectionPoint( dijon.EventMap, 'injector', dijon.Injector );
813 		this.injector.mapSingleton( dijon.EventMap );
814 		this.eventMap = this.injector.getInstance( dijon.EventMap );
815 	},
816 
817 	_wireActor : function(){
818 		this.injector.addInjectionPoint( dijon.Actor, 'eventDispatcher', dijon.EventDispatcher );
819 		this.injector.addInjectionPoint( dijon.Actor, 'eventMap', dijon.EventMap );
820 	},
821 
822 	_wireCommand : function(){
823 		this.injector.addInjectionPoint( dijon.Command, 'injector', dijon.Injector );
824 		//no need to map inherited properties, since dijon has covariant injections
825 	},
826 
827 	_wireAndCreateCommandMap : function(){
828 		this.injector.addInjectionPoint( dijon.CommandMap, 'eventMap', dijon.EventMap );
829 		this.injector.mapSingleton( dijon.CommandMap );
830 		this.commandMap = this.injector.getInstance( dijon.CommandMap );
831 	}
832 };//dijon.Context.prototype
