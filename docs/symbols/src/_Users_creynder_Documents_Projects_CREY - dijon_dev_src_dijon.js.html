 31 dijon.EventDispatcher.prototype = {
 32 
 33 	/**
 34 	 * @private
 35 	 * @param {String} eventType
 36 	 * @param {Number} index
 37 	 */
 38 	_removeListenerByIndex : function( eventType, index ){
 39 		this._listeners[ eventType ].splice( index, 1 );
 40 		if( this._listeners[ eventType ].length <= 0 )
 41 			delete this._listeners[ eventType ];
 42 		this._length--;
 43 	},
 44 
 45 	/**
 46 	 * adds a handler to be invoked when an event is dispatched
 47 	 * @param {String} eventType The name of the event to be listened to
 48 	 * @param {Function} listener The handler to be called when the event has been dispatched
 49 	 * @param {Boolean} [oneShot=false] Whether the listener must be called only once
 50 	 * @param {Boolean} [passEvent=false] Whether the event object needs to be passed to the listener
 51 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
 52 	*/
 53 	addListener : function( eventType, listener, oneShot, passEvent ) {
 54 		this.addScopedListener( eventType, listener, undefined, oneShot, passEvent );
 55 		return this;
 56 	},
 57 
 58 	/**
 59 	 * adds a handler to be invoked in a specific <code>scope</code> when an event of <code>eventType</code> is dispatched
 60 	 * @param {String} eventType The name of the event to be listened to
 61 	 * @param {Function} listener The handler to be called when the event has been dispatched
 62 	 * @param {Object} scope The scope in which the listener will be called
 63 	 * @param {Boolean} [oneShot=false] Whether the listener must be called only once
 64 	 * @param {Boolean} [passEvent=false] Whether the event object needs to be passed to the listener
 65 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
 66 	 */
 67 	addScopedListener : function( eventType, listener, scope, oneShot, passEvent ){
 68 		if( oneShot == undefined ) oneShot = false;
 69 		if( passEvent == undefined ) passEvent = false;
 70 		if( this._listeners[ eventType ] == undefined ){
 71 			this._listeners[ eventType ] = [];
 72 		}
 73 		this._listeners[ eventType ].push( {
 74 			scope : scope,
 75 			listener : listener,
 76 			oneShot : oneShot,
 77 			passEvent : passEvent
 78 		} );
 79 		++this._length;
 80 		return this;
 81 	},
 82 
 83 	/**
 84 	 * Checks whether the dispatcher object has registered <code>listener</code> for <code>eventType</code>
 85 	 * @param {String} eventType The name of the event to be listened to
 86 	 * @param {Function} listener The handler to be called when the event has been dispatched
 87 	 * @return {Boolean}
 88 	 */
 89 	hasListener : function( eventType, listener ){
 90 		return this.hasScopedListener( eventType, listener, undefined );
 91 	},
 92 
 93 	/**
 94 	 * Checks whether the dispatcher object has registered <code>listener</code> in <code>scope</code> for <code>eventType</code>
 95 	 * @param {String} eventType The name of the event to be listened to
 96 	 * @param {Function} listener The handler to be called when the event has been dispatched
 97 	 * @param {Object} scope The scope in which the listener will be called
 98 	 * @return {Boolean}
 99 	 */
100 	hasScopedListener : function( eventType, listener, scope ){
101 		if( this._listeners[ eventType ] ){
102 			var listeners = this._listeners[ eventType ];
103 			for( var i = 0, n = listeners.length ; i < n ; i++  ){
104 				var obj = listeners[ i ];
105 				if( obj.listener === listener && obj.scope === scope ){
106 					return true;
107 				}
108 			}
109 		}
110 
111 		return false;
112 	},
113 
114 	/**
115 	 * Removes a <code>listener</code> registered for <code>eventType</code>
116 	 * @param {String} eventType The name of the event to be listened to
117 	 * @param {Function} listener The handler to be called when the event has been dispatched
118 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
119 	 */
120 	removeListener : function( eventType, listener ){
121 		this.removeScopedListener( eventType, listener, undefined );
122 		return this;
123 	},
124 
125 	/**
126 	 * Removes a <code>listener</code> in <code>scope</code> registered for <code>eventType</code>
127 	 * @param {String} eventType The name of the event to be listened to
128 	 * @param {Function} listener The handler to be called when the event has been dispatched
129 	 * @param {Object} scope The scope in which the listener will be called
130 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
131 	 */
132 	removeScopedListener : function( eventType, listener, scope ) {
133 		for ( var i = 0 ; this._listeners[ eventType ] && i < this._listeners[ eventType ].length ;  ){
134 			var obj = this._listeners[ eventType ][ i ];
135 			if( obj.listener == listener && obj.scope === scope ){
136 				this._removeListenerByIndex( eventType, i );
137 			}else{
138 				i++;
139 			}
140 		}
141 		return this;
142 	},
143 
144 	/**
145 	 * dispatches an event with any number of arguments
146 	 * @param {Object|String} event The event object or event type
147 	 * @param {String} event.type if <code>event</code> is of type <code>Object</code> it requires a <code>type</code> property set to the name of the event that will be listened to.
148 	 * @param {...} [...=undefined] Any number of parameters
149 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
150 	 */
151 	dispatchEvent : function( event ){
152 		if (typeof event == "string"){
153 			event = { type: event };
154 		}
155 		if (!event.target){
156 			event.target = this;
157 		}
158 		var args = [];
159 		for( var i = 1, n = arguments.length; i < n ; i++ ){
160 			args.push( arguments[ i ] );
161 		}
162 
163 		if( this._listeners[ event.type ] ){
164 			for( var i = 0 ; this._listeners[ event.type ] && i < this._listeners[ event.type ].length ;  ){
165 				var obj = this._listeners[ event.type ][ i ];
166 				if( obj.oneShot ){
167 					this._removeListenerByIndex( event.type, i );
168 				}else{
169 					i++;
170 				}
171 				if( obj.passEvent ){
172 					args.unshift( event );
173 				}
174 				obj.listener.apply( obj.scope, args );
175 			}
176 		}
177 
178 		return this;
179 	},
180 
181 	/**
182 	 * removes all event listeners
183 	 * @return {dijon.EventDispatcher} The EventDispatcher instance
184 	 */
185 	removeAllListeners : function(){
186 	   this._listeners = {};
187 	   this._length = 0;
188 
189 	   return this;
190 	},
191 
192 	/**
193 	 * Returns the number of listeners
194 	 * @return {Number} The number of listeners
195 	 */
196 	length : function(){
197 		return this._length;
198 	}
199 
200 };//dijon.EventDispatcher.prototype

223 dijon.Dictionary.prototype = {
224 	/**
225 	 * @private
226 	 * @param key
227 	 * @return index
228 	 */
229 	_getIndexByKey : function( key ){
230 		for( var i = 0, n = this._map.length ; i < n ; i++ ){
231 			if( this._map[ i ].key === key ) return i;
232 		}
233 
234 		return -1;
235 	},
236 
237 	/**
238 	 * Maps <code>value</code> to <code>key</code>
239 	 * @param {Object} key
240 	 * @param {Object} value
241 	 * @return {dijon.Dictionary} the Dictionary instance
242 	 */
243 	add : function( key, value ){
244 		var index = this._getIndexByKey( key );
245 		if( index < 0 ){
246 			this._map.push( {
247 				key : key,
248 				value : value
249 			});
250 		}else{
251 			this._map[ index ] = {
252 				key : key,
253 				value : value
254 			}
255 		}
256 		return this;
257 	},
258 
259 	/**
260 	 * removes the mapping of the value of <code>key</code>
261 	 * @param {Object} key
262 	 * @return {dijon.Dictionary} the Dictionary instance
263 	 */
264 	remove : function( key ){
265 		var index = this._getIndexByKey( key );
266 		if( index >= 0 ) return this._map.splice( index, 1 ).value;
267 
268 		return this;
269 	},
270 
271 	/**
272 	 * retrieves the value mapped to <code>key</code>
273 	 * @param {Object} key
274 	 * @return {Object} the value mapped to <code>key</code>
275 	 */
276 	getValue : function( key ){
277 		var index = this._getIndexByKey( key );
278 
279 		if( index >= 0 ) return this._map[ index ].value;
280 
281 		return null;
282 	},
283 
284 	/**
285 	 * checks whether a value has been mapped to <code>key</code>
286 	 * @param {Object} key
287 	 * @return {Boolean}
288 	 */
289 	hasValue : function( key ){
290 		var index = this._getIndexByKey( key );
291 		return ( index >= 0 );
292 	}
293 
294 };//dijon.Dictionary.prototype

305 dijon.Injector = function(){
306 	this.qcn = 'dijon.Injector';
307 
308 	/** @private */
309 	this._mappingsByEventType = new dijon.Dictionary();
310 
311 	/** @private */
312 	this._injectionPoints = [];

314 
315 dijon.Injector.prototype = {
316 
317 	/**
318 	 * @private
319 	 * @param {Class} clazz
320 	 */
321 	_createAndSetupInstance : function( clazz ){
322 		var instance = new clazz();
323 		this.injectInto( instance );
324 		if( "setup" in instance ) instance.setup.call( instance );
325 		return instance;
326 	},
327 
328 	/**
329 	 * @private
330 	 * @param {Class} clazz
331 	 * @param {Boolean} overrideRules
332 	 * @return {Object}
333 	 */
334 	_retrieveFromCacheOrCreate : function( clazz, overrideRules ){
335 		var value = this._mappingsByEventType.getValue( clazz );
336 		var output = null;
337 		if( overrideRules ){
338 			output = this._createAndSetupInstance( clazz );
339 		}else{
340 			if( value ){
341 				if( value.isSingleton ){
342 					if( value.object == null  ){
343 						value.object = this._createAndSetupInstance( value.clazz );
344 					}
345 					output = value.object;
346 				}else{
347 					output = this._createAndSetupInstance( value.clazz );
348 				}
349 			}else{
350 				throw new Error( this.qcn + " is missing a rule for " + clazz );
351 			}
352 		}
353 		return output
354 	},
355 
356 	/**
357 	 * defines <code>propertyName</code> as an injection point for <code>targetClazz</code> to be injected with an instance
358 	 * of <code>sourceClazz</code>.
359 	 * @param {Class} targetClazz the class the injection point is applied to.
360 	 * @param {String} propertyName the <strong>name</strong> of the property used as an injection point.<br/>
361 	 * [!] MUST BE STRING
362 	 * @param {Class} sourceClazz the type of the instance that will be injected
363 	 */
364 	addInjectionPoint : function( targetClazz, propertyName, sourceClazz ){
365 		this._injectionPoints.push( {
366 			target : targetClazz,
367 			property : propertyName,
368 			clazz : sourceClazz
369 		} );
370 	},
371 
372 	/**
373 	 * Create (if possible) or retrieve an instance of the class mapped to <code>clazz</code>
374 	 * @param {Class} clazz
375 	 * @return {Object}
376 	 */
377 	getInstance : function( clazz ){
378 		return this._retrieveFromCacheOrCreate( clazz, false );
379 	},
380 
381 	/**
382 	 * When asked for an instance of the class <code>whenAskedFor<code> or object <code>whenAskedFor</code> inject an instance of <code>whenAskedFor<code>.
383 	 * @param {Class|Object} whenAskedFor
384 	 */
385 	mapSingleton : function( whenAskedFor ){
386 		if( this._mappingsByEventType.hasValue( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
387 		this.mapSingletonOf( whenAskedFor, whenAskedFor );
388 	},
389 
390 	/**
391 	 * When asked for an instance of the class <code>whenAskedFor</code> or object <code>whenAskedFor</code> inject the instance <code>useValue</code>.
392 	 * @param {Class|Object} whenAskedFor
393 	 * @param {Object} useValue
394 	 */
395 	mapValue : function( whenAskedFor, useValue ){
396 		if( this._mappingsByEventType.hasValue( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
397 		this._mappingsByEventType.add(
398 			whenAskedFor,
399 			{
400 				clazz : whenAskedFor,
401 				object : useValue,
402 				isSingleton : true
403 			}
404 		);
405 	},
406 
407 	/**
408 	 * Does a rule exist to satsify such a request?
409 	 * @param {Class} whenAskedFor
410 	 * @return {Boolean}
411 	 */
412 	hasMapping : function( whenAskedFor ){
413 		return this._mappingsByEventType.hasValue( whenAskedFor );
414 	},
415 
416 	/**
417 	 * When asked for an instance of the class <code>whenAskedFor</code> or for object <code>whenAskedFor</code> inject a <strong>new</strong> instance of <code>instantiateClass</code>.
418 	 * @param {Class|Object} whenAskedFor
419 	 * @param {Class} instantiateClass
420 	 */
421 	mapClass : function( whenAskedFor, instantiateClass ){
422 		if( this.hasMapping( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
423 		this._mappingsByEventType.add(
424 			whenAskedFor,
425 			{
426 				clazz : instantiateClass,
427 				object : null,
428 				isSingleton : false
429 			}
430 		);
431 	},
432 
433 	/**
434 	 * When asked for an instance of the class <code>whenAskedFor</code> or object <code>whenAskedFor</code> inject an instance of <code>useSingletonOf</code>.
435 	 * @param {Class|Object} whenAskedFor
436 	 * @param {Class} useSingletonOf
437 	 */
438 	mapSingletonOf : function( whenAskedFor, useSingletonOf ){
439 		if( this._mappingsByEventType.hasValue( whenAskedFor ) ) throw new Error( this.qcn + ' cannot remap ' + ' without unmapping first' );
440 		this._mappingsByEventType.add(
441 			whenAskedFor,
442 			{
443 				clazz : useSingletonOf,
444 				object : null,
445 				isSingleton : true
446 			}
447 		);
448 	},
449 
450 	/**
451 	 * create an instance of the class mapped to <code>clazz</code> and fulfill it's mapped dependencies<br/>
452 	 * <strong>WILL ALWAYS CREATE A NEW INSTANCE</strong>, even if <code>clazz</code> was mapped otherwise or not
453 	 * even mapped.
454 	 * @param {Class} clazz
455 	 * @return {Object}
456 	 */
457 	instantiate : function( clazz ){
458 		return this._retrieveFromCacheOrCreate( clazz, true );
459 	},
460 
461 	/**
462 	 * Perform an injection into an object, satisfying all it's dependencies
463 	 * @param {Object} instance
464 	 */
465 	injectInto : function( instance ){
466 		for( var i = 0, n = this._injectionPoints.length ; i < n ; i++ ){
467 			var mapping = this._injectionPoints[ i ];
468 			if( instance && instance instanceof mapping.target && mapping.property in instance )
469 				instance[ mapping.property ] = this.getInstance( mapping.clazz );
470 		}
471 	},
472 
473 	/**
474 	 * Remove a rule from the injector
475 	 * @param {Class|Object} whenAskedFor
476 	 */
477 	unmap : function( whenAskedFor ){
478 		this._mappingsByEventType.remove( whenAskedFor );
479 	},
480 
481 	/**
482 	 * removes an injection point mapping for a given class
483 	 * @param {Function} targetClazz
484 	 * @param {String} propertyName MUST BE STRING
485 	 * @see dijon.Injector#addInjectionPoint
486 	 */
487 	removeInjectionPoint : function( targetClazz, propertyName ){
488 		for( var i = 0, n = this._injectionPoints.length ; i < n ; i++ ){
489 			var point = this._injectionPoints[ i ];
490 			if( point.target == targetClazz && point.property == propertyName ) {
491 				this._injectionPoints.splice( i, 1 );
492 				return;
493 			}
494 		}
495 	}
496 
497 };//dijon.Injector.prototype

503 /**
504  * registers class members as listeners for specific events, before (or after) instantiation.<br/>
505  * Allows for mapping listeners to lazily instantiated objects.<br/>
506  * [!] This class differs substantially from the RobotLegs EventMap both in use and functionality
507  * @class dijon.EventMap
508  * @author Camille Reynders - www.creynders.be
509  * @constructor
510  */
511 dijon.EventMap = function(){
512 	this.qcn = 'dijon.EventMap';
513 
514 	/**
515 	 * @private
516 	 * @type Object
517 	 */
518 	this._mappingsByEventType = {};
519 
520 	/**
521 	 * @private
522 	 * @type Object
523 	 */
524 	this._mappingsNumByClazz = {};
525 
526 	/**
527 	 * @private
528 	 * @type dijon.EventDispatcher
529 	 */
530 	this.dispatcher = undefined; //inject
531 
532 	/**
533 	 * @private
534 	 * @type dijon.Injector
535 	 */
536 	this.injector = undefined; //inject
537 
538 };//dijon.EventMap
539 
540 dijon.EventMap.prototype = {
541 
542 	/**
543 	 * @private
544 	 * @param eventType
545 	 * @param clazz
546 	 * @param handler
547 	 */
548 	_getMappingIndex : function( mappingsListForEvent, clazz, handler ){
549 		if( mappingsListForEvent ){
550 			for( var i = 0, n = mappingsListForEvent.length; i < n ; i++ ){
551 				var mapping = mappingsListForEvent[ i ];
552 				if( mapping.clazz === clazz && mapping.handler === handler ){
553 					return i;
554 				}
555 			}
556 		}
557 
558 		return -1;
559 	},
560 				
561 	/**
562 	 * @private
563 	 * @param {Object} event
564 	 */
565 	_handleRuledMappedEvent : function( event ){
566 		var mappingsListForEvent = this._mappingsByEventType[ event.type ];
567 		var args = [];
568 
569 		for( var i = 1, n = arguments.length; i < n ; i++ ){
570 			args.push( arguments[ i ] );
571 		}
572 		for( var i = 0, n = mappingsListForEvent.length; i < n; i++ ){
573 			var obj = mappingsListForEvent[i];
574 			if( this.injector.hasMapping( obj.clazz ) ){
575 				if( obj.oneShot )
576 					this.removeRuledMapping( event.type, obj.clazz, obj.handler );
577 				var instance = this.injector.getInstance( obj.clazz );
578 				if( obj.passEvent )
579 					args.unshift( event );
580 				if( obj.handler != null )
581 					obj.handler.apply( instance, args );
582 			}else{
583 				//injector mapping has been deleted, but
584 				//eventMap mapping not
585 				//TODO: remove or throw error?
586 			}
587 		}
588 	},
589 
590 	/**
591 	 * @private
592 	 * @param {String} eventType
593 	 * @param {Class} clazz
594 	 * @param {Function} handler
595 	 */
596 	_removeRuledMappingAndUnmapFromInjectorIfNecessary : function( eventType, clazz, handler ){
597 		this.removeRuledMapping( eventType, clazz, handler );
598 		var mappingsNum = this._mappingsNumByClazz[ clazz ] || 0;
599 		if( mappingsNum <= 0 )
600 			this.injector.unmap( clazz );
601 	},
602 
603 	/**
604 	 * maps <code>handler</code> as a listener for <code>eventType</code> to be called as a member of a <code>clazz</code>
605 	 * instance. The instance will be created according to the rule defined for <code>clazz</code> in injector.
606 	 * <br/>[!] requires <code>clazz</code> is already ruled by the injector
607 	 * @see dijon.Injector
608 	 * @param {String} eventType The name of the event to be listened to
609 	 * @param {Class} clazz
610 	 * @param {Function} handler
611 	 * @param {Boolean} [oneShot=false] Whether the listener must be called only once
612 	 * @param {Boolean} [passEvent=false] Whether the event object should be passed as a parameter to <code>handler</code>
613 	 * upon invocation or not. If <code>true</code> any additional dispatched values will be passed as parameters after
614 	 * the event object
615 	 */
616 	addRuledMapping : function( eventType, clazz, handler, oneShot, passEvent ){
617 		if( ! this._mappingsByEventType[ eventType ] ){
618 			this._mappingsByEventType[ eventType ] = [];
619 			this.dispatcher.addScopedListener( eventType, this._handleRuledMappedEvent, this, false, true );
620 		}
621 
622 		var mappingsNum = this._mappingsNumByClazz[ clazz ] || 0;
623 		this._mappingsNumByClazz[ clazz ] = ++mappingsNum;
624 
625 		this._mappingsByEventType[ eventType ].push( { clazz : clazz, handler : handler, oneShot : oneShot, passEvent: passEvent } );
626 	},
627 
628 	/**
629 	 * Removes the mapping for <code>clazz</code>
630 	 * @see dijon.EventMap#addRuledMapping
631 	 * @param {String} eventType The name of the event to be listened to
632 	 * @param {Class} clazz
633 	 * @param {Function} handler
634 	 */
635 	removeRuledMapping : function( eventType, clazz, handler ){
636 		var mappingsListForEvent = this._mappingsByEventType[ eventType ];
637 		var index = this._getMappingIndex( mappingsListForEvent, clazz, handler );
638 		if( index >= 0 ){
639 			var mapping = mappingsListForEvent[ index ];
640 			delete mapping.clazz;
641 			delete mapping.handler;
642 			mapping = null;
643 			mappingsListForEvent.splice( index, 1 );
644 			if( mappingsListForEvent.length <= 0 )
645 				delete mappingsListForEvent[ eventType ];
646 			var mappingsNum = this._mappingsNumByClazz[ clazz ] || 0;
647 			if( mappingsNum <= 0 ){
648 				delete this._mappingsNumByClazz[ clazz ];
649 			}else{
650 				this._mappingsNumByClazz[ clazz ] = --mappingsNum;
651 			}
652 			return true;
653 
654 		}
655 		return false;
656 	},
657 
658 	/**
659 	 *
660 	 * @param {String} eventType
661 	 * @param {Class} clazz
662 	 * @param {Function} [handler=null]
663 	 * @param {Boolean} [oneShot=false]
664 	 * @param {Boolean} [passEvent=false]
665 	 */
666 	addClassMapping : function( eventType, clazz, handler, oneShot, passEvent ){
667 		if( ! this.injector.hasMapping( clazz ) )
668 			this.injector.mapClass( clazz, clazz );
669 		if( handler == undefined )
670 			handler = null;
671 		this.addRuledMapping( eventType, clazz, handler, oneShot, passEvent );
672 	},
673 
674 	/**
675 	 *
676 	 * @param {String} eventType
677 	 * @param {Class} clazz
678 	 * @param {Function} [handler=null]

681 		if( handler == undefined )
682 			handler = null;
683 		this._removeRuledMappingAndUnmapFromInjectorIfNecessary( eventType, clazz, handler );
684 	},

686 	/**
687 	 *
688 	 * @param {String} eventType
689 	 * @param {Class} clazz
690 	 * @param {Function} [handler=null]
691 	 */
692 	hasMapping : function( eventType, clazz, handler ){
693 		if( handler == undefined )
694 			handler = null;
695 		return this._getMappingIndex( this._mappingsByEventType[ eventType ], clazz, handler ) >= 0 ;
696 	}

830  * @constructor

832 dijon.Context = function(){
833 	this.qcn = 'dijon.Context';
834 
835 	/**
836 	 * @type dijon.Injector
837 	 */
838 	this.injector = undefined;
839 
840 	/**
841 	 * @type dijon.EventMap
842 	 */
843 	this.eventMap = undefined;
844 
845 	/**
846 	 * @type dijon.EventDispatcher
847 	 */
848 	this.eventDispatcher = undefined;
849 
850 	/**
851 	 * @type dijon.CommandMap
852 	 */
853 	this.commandMap = undefined;
854 
855 	this._createInjector();
856 	this._wireAndCreateEventDispatcher();
857 	this._wireAndCreateEventMap();
858 	this._wireActor();
859 	this._wireCommand();
860 	this._wireAndCreateCommandMap();

864 dijon.Context.prototype = {
865 
866 	/**
867 	 * @private
868 	 */
869 	_createInjector : function(){
870 		this.injector = new dijon.Injector();
871 		this.injector.mapValue( dijon.Injector, this.injector );
872 	},
873 	/**
874 	 * @private
875 	 */
876 	_wireAndCreateEventDispatcher : function(){
877 		this.injector.mapSingleton( dijon.EventDispatcher );
878 		this.eventDispatcher = this.injector.getInstance( dijon.EventDispatcher );
879 	},
880 	/**
881 	 * @private
882 	 */
883 	_wireAndCreateEventMap : function(){
884 		this.injector.addInjectionPoint( dijon.EventMap, 'dispatcher', dijon.EventDispatcher );
885 		this.injector.addInjectionPoint( dijon.EventMap, 'injector', dijon.Injector );
886 		this.injector.mapSingleton( dijon.EventMap );
887 		this.eventMap = this.injector.getInstance( dijon.EventMap );
888 	},
889 
890 	_wireActor : function(){
891 		this.injector.addInjectionPoint( dijon.Actor, 'eventDispatcher', dijon.EventDispatcher );
892 		this.injector.addInjectionPoint( dijon.Actor, 'eventMap', dijon.EventMap );
893 	},
894 
895 	_wireCommand : function(){
896 		this.injector.addInjectionPoint( dijon.Command, 'injector', dijon.Injector );
897 		this.injector.addInjectionPoint( dijon.Command, 'commandMap', dijon.CommandMap );
898 		//no need to map inherited properties, since dijon has covariant injections
899 	},
900 
901 	_wireAndCreateCommandMap : function(){
902 		this.injector.addInjectionPoint( dijon.CommandMap, 'eventMap', dijon.EventMap );
903 		this.injector.addInjectionPoint( dijon.CommandMap, 'injector', dijon.Injector );
904 		this.injector.mapSingleton( dijon.CommandMap );
905 		this.commandMap = this.injector.getInstance( dijon.CommandMap );
906 	}
907 };//dijon.Context.prototype
